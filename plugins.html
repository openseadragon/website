A plugin is any object that takes a viewer and returns a destroy function. The seams: tile-loaded, animation, canvas-click, and the drawer hook chain.

my-scalebar-plugin.js esm 
export function scalebar(viewer, opts = {}) {
  const el = document.createElement("div");
  el.className = "scalebar";
  viewer.canvas.appendChild(el);

  const redraw = () => {
    const mpp = opts.micronsPerPixel;
    const px = viewer.viewport.deltaPixelsFromPoints(new OpenSeadragon.Point(1, 0)).x;
    el.textContent = (px * mpp).toFixed(1) + " µm";
  };

  viewer.addHandler("animation", redraw);
  redraw();

  return () => {
    viewer.removeHandler("animation", redraw);
    el.remove();
  };
}
